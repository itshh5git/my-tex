\begin{definition}
    Let $X$ be topological space.
    A \textbf{curve} in $X$ is an continuous map $\alpha: \mathrm{I} \rightarrow X$ of an open interval $\mathrm{I}=(a, b)$ of the real line $\mathbb{R}$ into $X$
\end{definition}

\begin{definition}
    A \textbf{parametrized differentiable curve} is a differentiable map $\alpha: \mathrm{I} \rightarrow \mathbb{R}^3$
\end{definition}

\begin{definition}
    A parametrized differentiable curve $\alpha: \mathrm{I} \rightarrow \mathbb{R}^3$ is said to be \textbf{regular} if $\alpha^{\prime}(t) \neq 0$ for all $t \in \mathrm{I}$.
\end{definition}

\begin{definition}
    Let $\alpha: \mathrm{I} \rightarrow \mathbb{R}^3$ be a curve parametrized by arc length $s \in \mathrm{I}$ without singular points of order $<2$ (i.e., $\alpha^{\prime}(s) \neq 0$ and $\alpha^{\prime \prime}(s) \neq 0$ for all $s \in \mathrm{I}$). Then we define:
    \begin{enumerate}
        \item the unit vector $\mathbf{t}(s):=\alpha^{\prime}(s)$ is called the \textbf{tangent vector} of $\alpha$ at $s$.
        \item
              The number $\kappa(s):=\left|\alpha^{\prime \prime}(s)\right|$ is called the \textbf{curvature} of $\alpha$ at $s$.
              The unit vector $\mathbf{n}(s):=\frac{\alpha^{\prime \prime}(s)}{\kappa(s)}$ is called the \textbf{principal normal vector} of $\alpha$ at $s$.
        \item
              the unit vector $\mathbf{b}(s):=\mathbf{t}(s) \times \mathbf{n}(s)$ is called the \textbf{binormal vector} of $\alpha$ at $s$.
              The number $\tau(s):=\mathbf{b}^{\prime}(s) \cdot \mathbf{n}(s)$ is called the \textbf{torsion} of $\alpha$ at $s$.
    \end{enumerate}
\end{definition}

\begin{definition}
    \begin{enumerate}
        \item
              The plane determined by $\alpha^{\prime}(s)$ and $n(s)$, is called the \textbf{osculating plane} at $s$.
        \item
              The plane determined by $\mathbf{n}(s)$ and $\mathbf{b}(s)$ is called the \textbf{normal plane} at $s$.
        \item
              The plane determined by $\alpha^{\prime}(s)$ and $\mathbf{b}(s)$ is called the \textbf{rectifying plane} at $s$.
    \end{enumerate}
\end{definition}
\begin{theorem}
    [Frenet-Serret Formulas]
    Let $\alpha: \mathrm{I} \rightarrow \mathbb{R}^3$ be a curve parametrized by arc length $s \in \mathrm{I}$ without singular points of order $<2$. Then the following formulas hold:
    \begin{equation*}
        \left(\begin{matrix}
            \mathbf{t}^\prime \\
            \mathbf{n}^\prime \\
            \mathbf{b}^\prime \\
        \end{matrix}\right)
        =
        \left(\begin{matrix}
                     & \kappa &       \\
            - \kappa &        & -\tau \\
                     & \tau   &       \\
        \end{matrix}\right)
        \left(\begin{matrix}
            \mathbf{t} \\
            \mathbf{n} \\
            \mathbf{b} \\
        \end{matrix}\right)
    \end{equation*}
\end{theorem}

\begin{proposition}
    If $f: U \rightarrow \mathbb{R}$ is a differentiable function in an open set $U$ of $\mathbb{R}^2$, then the graph of $f$ , that is, the subset of $\mathbb{R}^3$ given by ( $\mathrm{x}, \mathrm{y}, \mathrm{f}(\mathrm{x}, \mathrm{y})$ ) for $(\mathrm{x}, \mathrm{y}) \in \mathrm{U}$, is a regular surface.
\end{proposition}






\begin{proposition}
    If $f: U \subset \mathbb{R}^3 \rightarrow \mathbb{R}$ is a differentiable function and $a\in f(U)$ is a regular value of $f$, then $f^{-1}(a)$ is a regular surface in $\mathbb{R}^3$.
\end{proposition}





\section{First Fundamental Form}
\begin{definition}
    Let $S$ be a regular surface in $\mathbb{R}^3$ and $p \in S$.
    The natural inner product of $R^3 \supset S$ induces on each tangent plane $T_p(S)$ of a regular surface $S$ an inner product, to be denoted by $\langle,\rangle_p$ : If $w_1, w_2 \in T_p(S) \subset R^3$, then $\left\langle w_1, w_2\right\rangle_p$ is equal to the inner product of $w_1$ and $w_2$ as vectors in $R^3$. To this inner product, which is a symmetric bilinear form


    (i.e., $\left\langle w_1, w_2\right\rangle=\left\langle w_2, w_1\right\rangle$ and $\left\langle w_1, w_2\right\rangle$ is linear in both $w_1$ and $w_2$ ), there corresponds a quadratic form $I_p: T_p(S) \rightarrow R$ given by

    \begin{equation*}
        I_p(w)=\langle w, w\rangle_p=|w|^2 \geq 0 .
    \end{equation*}

    The quadratic form $\mathrm{I}_{\mathrm{p}}$ on $\mathrm{T}_{\mathrm{p}}(\mathrm{S})$, defined by Eq. (1), is called the \textbf{first fundamental form} of $S$ at $\mathrm{p} \in S$.
    \begin{remark}
        We shall now express the first fundamental form in the basis $\left\{\mathbf{x}_u, \mathbf{x}_v\right\}$ associated to a parametrization $\mathbf{x}(u, v)$ at $p$. Since a tangent vector $w \in T_p(S)$ is the tangent vector to a parametrized curve $\alpha(t)=\mathbf{x}(u(t), v(t)), t \in(-\epsilon, \epsilon)$, with $p=\alpha(0)=\mathbf{x}\left(u_0, v_0\right)$, we obtain

        \begin{equation*}
            \begin{aligned}
                I_p\left(\alpha^{\prime}(0)\right) & =\left\langle\alpha^{\prime}(0), \alpha^{\prime}(0)\right\rangle_p                                                                                                                                                                          \\
                                                   & =\left\langle\mathbf{x}_u u^{\prime}+\mathbf{x}_v v^{\prime}, \mathbf{x}_u u^{\prime}+\mathbf{x}_v v^{\prime}\right\rangle_p                                                                                                                \\
                                                   & =E\left(u^{\prime}\right)^2+2 F u^{\prime} v^{\prime}+G\left(v^{\prime}\right)^2
            \end{aligned}
        \end{equation*}
        where the values of the functions involved are computed for $t=0$, and
        \begin{equation*}
            \begin{aligned}
                 & E\left(u_0, v_0\right)=\left\langle\mathbf{x}_u, \mathbf{x}_u\right\rangle_p, \\
                 & F\left(u_0, v_0\right)=\left\langle\mathbf{x}_u, \mathbf{x}_v\right\rangle_p, \\
                 & G\left(u_0, v_0\right)=\left\langle\mathbf{x}_v, \mathbf{x}_v\right\rangle_p
            \end{aligned}
        \end{equation*}
        are the coefficients of the first fundamental form in the basis $\left\{\mathbf{x}_u, \mathbf{x}_v\right\}$ of $T_p(S)$.
        
         By letting $p$ run in the coordinate neighborhood corresponding to $\mathbf{x}(u, v)$ we obtain functions $E(u, v), F(u, v), G(u, v)$ which are differentiable in that neighborhood.
    \end{remark}
\end{definition}
