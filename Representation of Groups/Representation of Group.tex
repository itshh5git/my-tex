\chapter{Representation of Groups and Group Algebras}

\section{Basic Definitions}

\begin{definition}
    Suppose now $G$ is a finite group and $k$ is a field.
    \begin{enumerate}
        \item
              A \textbf{$k$-linear representation} of $G$
              is a group homomorphism
              \begin{equation*}
                  \rho : G \rightarrow\GL(V)
              \end{equation*}
              where $V$ is a $k$-vector space.
              The space $V$ is called the \textbf{representation space} of $\rho$ and $\dim_k V$ (if finite) is called the \textbf{degree} of the representation.

        \item
              Let $\rho$ and $\rho^{\prime}$ be two representations of the same group $G$ in vector spaces $V$ and $V^{\prime}$.
              The \textbf{transforms} of $V$ and $V^\prime$ is a $k$-linear map from $V$ to $V^{\prime}$ such that
              \begin{equation*}
                  \tau \circ \rho(s)=\rho^{\prime}(s) \circ \tau \quad \text { for all } s \in G .
              \end{equation*}
              These representations are said to be \textbf{similar} (or isomorphic) if $\tau$ is an isomorphism.
    \end{enumerate}
    \begin{remark}
        A $k$-linear representation $\rho : G \rightarrow\GL(V)$ is equivalent to a left $k[G]$-module $V$.
        In this case, the transforms between two representations are exactly the $k[G]$-module homomorphisms.
    \end{remark}
\end{definition}

\begin{definition}
    Let $\rho : G \rightarrow\GL(V)$ be a $k$-linear representation of a finite group $G$ in the vector space $V$.
    \begin{enumerate}
        \item
              A subspace $W$ of $V$ is called a \textbf{subrepresentation} if it is a $k[G]$-submodule of $V$.

        \item
              A representation is called \textbf{irreducible} if it has no proper non-zero subrepresentation.
    \end{enumerate}
\end{definition}


\begin{definition}
    Let $V_i$ be $k$-linear representation of a finite group $G$ .
    \begin{enumerate}
        \item
              The \textbf{sum representation} $V_1 \oplus V_2$ is defined on the vector space $V_1 \oplus V_2$ by
              \begin{equation*}
                  g\left(v_1, v_2\right)=\left(g v_1, g v_2\right)
              \end{equation*}
              for all $g \in G, v_1 \in V_1, v_2 \in V_2$.
              (the direct sum of $k[G]$-modules $V_1 \oplus V_2$).
        \item
              The \textbf{tensor product representation} $V_1 \otimes V_2$ is defined on the vector space $V_1 \otimes V_2$ by
              \begin{equation*}
                  g\left(v_1 \otimes v_2\right)=\left(g v_1\right) \otimes\left(g v_2\right)
              \end{equation*}
              for all $g \in G, v_1 \in V_1, v_2 \in V_2$. (inner tensor product of $k[G]$ module $V_1 V_2$)
        \item
              The $n$-th tensor power $V^{\otimes n}$ is again a representation of $G$ by this rule, and the exterior powers $\bigwedge^n(V)$ and symmetric powers $\operatorname{Sym}^n(V)$ are subrepresentations of it.

        \item
              The dual $V^*=\operatorname{Hom}_k(V, k)$ of $V$ is also a representation of $G$ by the rule
              \begin{equation*}
                  \left\langle g\cdot f, \square \right\rangle
                  =
                  \left\langle f,g^{-1}\square \right\rangle
              \end{equation*}
    \end{enumerate}
\end{definition}

\begin{proposition}
    For representations $V,U,W$ of $G$, we have
    \begin{equation*}
        \begin{aligned}
            V \otimes(U \oplus W)       & =(V \otimes U) \oplus(V \otimes W),                     \\
            \bigwedge^k(V \oplus W)     & =\bigoplus_{a+b=k} \bigwedge^a V \otimes \bigwedge^b W, \\
            \bigwedge^k\left(V^*\right) & =\bigwedge^k(V)^*,
        \end{aligned}
    \end{equation*}
\end{proposition}



\section{Maschke's Theorem}
In this section, we always assume that $k$ be a field whose characteristic does not divide the order of the finite group $G$
\begin{lemma}
    Let $V_i$ be $k[G]$-module for $i=1,2$ and $f:V_1\rightarrow V_2$ be a $k$-linear map, then
    \begin{equation*}
        F
        :=
        \frac{1}{\left|G\right|} \sum_{g \in G} \rho_g^2 \cdot f \cdot \left(\rho_g^1\right)^{-1}
    \end{equation*}
    is a $k[G]$-module homomorphism from $V_1$ to $V_2$, where $1/\left|G\right|$ is the inverse of the order of $G$ in the field $k$
\end{lemma}

\begin{theorem}[Maschke]
    If the characteristic of the field $k$ does not divide the $\left|G\right|$, then $k[G]$ is semisimple.
    \begin{proof}
        Let $V$ be a $k[G]$-module, $W$ be a submodule of $V$. And let $W^{\prime}$ be a $k$-complement of $W$ in $V$, and let $p$ be the corresponding projection of $V$ onto $V$.
        Define a map $P: V \rightarrow W$ by
        \begin{equation*}
            P(v)
            :=
            \frac{1}{\left|G\right|} \sum_{g \in G} g \cdot p\left( g^{-1}v\right)
        \end{equation*}
        Then $P$ is a $k[G]$-projection from $V$ to $W$ (i.e. $P$ is a $k[G]$-module homomorphism with $P^2 = P$ and $W=\operatorname{Im}(P)$), so $\ker(P)$ is a $k[G]$-complement of $W$ in $V$.
    \end{proof}
\end{theorem}

\begin{corollary}
    It follows from $\left|G\right|\leq \infty$ that $k[G]$ is a semi-simple Artinian algebra. Thus the regular representation $R_G:={}_{k[G]} k[G]$ has a unique decomposition
    \begin{equation*}
        R_G\cong \bigoplus_{i=1}^m V_i^{\oplus n_i}
    \end{equation*}
    where $V_i$ are all irreducible representations of $G$ non-isomorphic to each other, and $n_i$ is the multiplicity of $V_i$ in $R_G$. ($n_i=\# \Hom_{k[G]}\left( R_G,V_i \right)$)

    Every representation of $G$ is a direct sum of $V_i$'s.(that is why we only need to study irreducible representations of finite degree)
\end{corollary}


\begin{corollary}
    The dimension of an irreducible representation of $G$ divides the order of $G$.
\end{corollary}

\section{Characters of Representations}
\begin{definition}
    Let $G$ be a group and $V$ be a $k[G]$-module of finite dimension (or say $V$ is a representation of $G$).
    \begin{enumerate}
        \item
              The \textbf{character} $\chi$ of the representation is defined by
              \begin{equation*}
                  \chi(s)
                  :=
                  \operatorname{Tr}\left(\rho_s\right) .
              \end{equation*}
        \item
              If $W$ be a submodule of $V$, the \textbf{subcharacter} $\chi_{W}$ of $W$ is defined by
              \begin{equation*}
                  \chi_{W}(s)
                  :=
                  \operatorname{Tr}\left(\rho_s|_{W}\right) .
              \end{equation*}
        \item
              A character $\chi$ is called \textbf{irreducible} if it is the character of an irreducible representation.
    \end{enumerate}
\end{definition}



\begin{proposition}
    If $\chi$ is the character of a representation $V$ of degree $n$, we have:
    \begin{enumerate}
        \item
              $\chi(1)=n$,
        \item
              $\chi\left(s^{-1}\right)=\overline{\chi(s)}$ for $s \in G$,
        \item
              $\chi\left(t s t^{-1}\right)=\chi(s)$ for $s, t \in G$. $\chi(a b)=\chi(b a)$
    \end{enumerate}
\end{proposition}
\begin{proposition}
    Let $V_1,V_2$ be two linear representations of $G$.
    Then:
    \begin{enumerate}
        \item
              $\chi_{V_1 \oplus V_2} = \chi_{V_1} + \chi_{V_2}$.
        \item
              $\chi_{V_1\otimes V_2} = \chi_{V_1} \cdot \chi_{V_2}$.
        \item
              $\chi_{\bigwedge^2 V}\left(g\right) = \frac{1}{2}\left(\chi_V(g)^2 - \chi_(g^2)\right)$.
        \item
              $\chi_{\operatorname{Sym}^2 V}\left(g\right) = \frac{1}{2}\left(\chi_V(g)^2 + \chi_(g^2)\right)$.
    \end{enumerate}
\end{proposition}

\section{Schur's lemma and }% Schur's lemma

\begin{definition}
    Let $f,g$ be $k$-valued functions on a finite group $G$.
    The \textbf{inner product} of $f$ and $g$ is defined
    by
    \begin{equation*}
        \left(f ,g \right)
        :=
        \frac{1}{\left|G\right|} \sum_{s \in G} f(s) g(s^{-1}).
    \end{equation*}
\end{definition}

\begin{proposition}[Schur's lemma]
    Let $k$ be a algebraically closed field and $G$ be a finite group.
    Let $\rho^1: G \rightarrow \GL\left(V_1\right)$ and $\rho^2: G \rightarrow \GL\left(V_2\right)$ be two irreducible representations of $G$, and let $f:V_1 \rightarrow V_2$ be a transform.
    Then:
    \begin{enumerate}
        \item
              If $\rho^1$ and $\rho^2$ are not isomorphic, we have $f=0$.

        \item
              If $\rho^1 \cong \rho^2$ ($\rho^1= \rho^2$ and $V_1 =   V_2$ ) then
              $f$ is a homothety (i.e., a scalar multiple of the identity).
    \end{enumerate}
\end{proposition}

\begin{corollary}
    Let $h$ be a linear mapping of $V_1$ into $V_2$, and put:
    \begin{equation*}
        h^0=\frac{1}{\left|G\right|} \sum_{g \in G}\left(\rho_g^2\right)^{-1} h \rho_g^1 .
    \end{equation*}
    Then:
    \begin{enumerate}
        \item
              If $\rho^1$ and $\rho^2$ are not isomorphic, we have $h^0=0$. In this case,
              \begin{equation*}
                  \left(r_{i_2j_2},r_{i_2j_2}
                  \right)
                  =
                  \frac{1}{\left|G\right|}\sum_{g\in G} r_{i_2j_2}(g^{-1}) r_{i_1j_1}(g) =0
              \end{equation*}
              for arbitary $i_1,i_2,j_1,j_2$.

        \item
              If $V_1=V_2$ and $\rho^1=\rho^2, h^0$ is a homothety of ratio $(1 / n) \operatorname{Tr}(h)$, with $n=\operatorname{dim}\left(V_1\right)$.
              In this case,
              \begin{equation*}
                  \left(r_{i_2j_2},r_{i_2j_2}
                  \right)
                  =
                  \frac{1}{\left|G\right|}\sum_{g\in G} r_{i_2j_2}(g^{-1}) r_{i_1j_1}(g)
                  =
                  \frac{1}{n} \delta_{i_1 i_2}\cdot \delta_{j_1 j_2}
              \end{equation*}
    \end{enumerate}
\end{corollary}

        \item  $\left(\chi,\chi \right)=\sum_{k=1}^h m_k^2$

\begin{proposition}
    The character $r_G$ of the regular representation of $G$ is given by
    \begin{equation*}
        r_G(g)
        =
        \begin{cases}
            \left|G\right| & ,g=e      \\
            0              & ,g \neq e \\

    \end{equation*}
\end{proposition}
\begin{corollary}

        \item

        \item

    \end{enumerate}
\end{corollary}

\begin{lemma}
    Let $f:G\rightarrow k$ be any function on $G$, and for a representation $V$ of $G$ set
    \begin{equation*}
        F
        :=
        \sum_{g \in G} f(g) \cdot \rho_g
    \end{equation*}
    Then $F$ is a $k[G]$-module homomorphism for all $V$ if and only if $f$ is a class function on $G$.
\end{lemma}
\begin{corollary}
    If $\rho:G \to \GL(V)$ is an irreducible representation with character $\chi$, then $F$ is a homothety of ratio
    \begin{equation*}
        \lambda=
        \frac{1}{n}\sum_{g \in G} f(g) \chi(g)
        =
        \frac{\left|G\right|}{n} \left(f,\overline{\chi} \right)
    \end{equation*}
    where $n=\dim V$.
\end{corollary}


\begin{theorem}
    The all characters $\chi_1,\chi_2,\cdots ,\chi_h$ form an orthonormal basis of the space $\mathcal{H}$ of class functions on $G$ with values in $k$.

    The number of non-isomorphic irreducible representations of $G$ is equal to the number of conjugacy classes of $G$.
\end{theorem}

\begin{proposition}
    Let $s \in G$, and let $c(s)$ be the number of elements in the conjugacy class of $s$. Then
    \begin{enumerate}
        \item
              the class function equal to $1$ on the class of $s$ and equal to 0 elsewhere is given by
              \begin{equation*}
                  f_s(t)=\frac{c(s)}{|G| } \sum_{i=1}^{i=h} \chi_i(s^{-1}) \chi_i(t)
              \end{equation*}
        \item
              For $t \in G$ not conjugate to $s$, we have

                  \end{cases}
              \end{equation*}
    \end{enumerate}
\end{proposition}

\begin{definition}

              \begin{equation*}
                  \operatorname{Ind}_H^G W
                  :=
                  k[G] \otimes_{k[H]} W
              \end{equation*}
              as left $k[G]$-module.
              \begin{remark}[Mackey's definition]
                  Let $\left\{g_i \mid i \in I\right\}$ be a complete set of representatives of the left cosets of $H$ in $G$.
                  Then as a $k$-vector space,
                  \begin{equation*}
                      \operatorname{Ind}_H^G W
                      \cong
                      \bigoplus_{i \in I} g_i \otimes W
                  \end{equation*}
                  and the action of $G$ is given by
                  \begin{equation*}
                      s \cdot\left(g_i \otimes w\right)
                      =\left(s \cdot g_i\right) \otimes w
                      =\left( g_j\cdot h\right) \otimes w
                      =
                      g_j \otimes \left(h \cdot w\right)
                  \end{equation*}
                  where $s g_i=g_j h$ with $h \in H$.
              \end{remark}
        \item
              If $\left(\pi,V\right)$ is a representation of $G$. Define the \textbf{restricted representation} $\operatorname{Res}_H^G V$ of $H$ by

              \end{remark}

\section{Frobenius reciprocity}
\begin{definition}
    If $f$ is a class function on $H$ with values in $k$, consider the
    \begin{equation*}
        \operatorname{Ind}_H^G  f(s)
        :=
        \frac{1}{\left|H\right|}\sum_{\substack{g\in G\\g^{-1}sg \in H}} f\left(g^{-1}sg\right)
    \end{equation*}
    called the \textbf{induced class function} of $f$ from $H$ to $G$, denoted by $\operatorname{Ind}_H^G f$.
    Then:
    \begin{enumerate}
        \item $\operatorname{Ind}_H^G f $ is a class function on $G$.
        \item If $\chi$ is the character of a representation of $H$, then $\operatorname{Ind}_H^G \chi$ is the character of the induced representation $\operatorname{Ind}_H^G W$.
    \end{enumerate}
\end{definition}

\begin{definition}
    If $f$ is a class function on $G$ with values in $k$, consider the
    \begin{equation*}
        \operatorname{Res}_H^G  f(s)
        :=
        f(s)
    \end{equation*}
    called the \textbf{restricted class function} of $f$ from $G$ to $H$, denoted by $\operatorname{Res}_H^G f$.
    Then:
    \begin{enumerate}
        \item $\operatorname{Res}_H^G f $ is a class function on $H$.
        \item If $\chi$ is the character of a representation of $G$, then $\operatorname{Res}_H^G \chi$ is the character of the restricted representation $\operatorname{Res}_H^G V$.
    \end{enumerate}
\end{definition}



\begin{lemma}
    Let $G$ be a group and $\chi_i$ be the character of $V_i$,
    \begin{equation*}
        \left(\chi_2 , \chi_2 \right)
        =
        \left(V_1,V_2 \right)
        :=
        \dim_k \Hom_{k[G]}\left( V_1 ,V_2 \right)
    \end{equation*}
\end{lemma}


\begin{theorem}[Frobenius reciprocity]
    Given a $k[G]$-module $V$, and a $k[H]$-module $W$, we have
    \begin{equation*}
        \Hom_{k[H]}\left(W  ,\Res_H^G V \right)
        \cong \Hom_{k[G]}\left( \operatorname{Ind}_H^G W , V \right)
    \end{equation*}
    as $k$-vector spaces.
\end{theorem}

\begin{theorem}[Frobenius reciprocity]
    Let $H$ be a subgroup of a finite group $G$.
    If $\chi$ is the character of a representation of $H$ and $\psi$ is the character of a representation of $G$, then
    \begin{equation*}
        \left(\operatorname{Ind}_H^G \chi ,\psi \right)_G
        =
        \left(\chi ,\operatorname{Res}_H^G \psi \right)_H
    \end{equation*}
    Thus it is true for class functions.
    \begin{proof}
        Let $W$ be the representation space of $\chi$ and $V$ be the representation space of $\psi$.
        By the definition of inner product and the previous lemma, we have
        \begin{equation*}
            \left(\operatorname{Ind}_H^G \chi ,\psi \right)_G
            =
            \dim_k \Hom_{k[G]}\left( \operatorname{Ind}_H^G W ,V \right)
            =
            \dim_k \Hom_{k[H]}\left( W ,\operatorname{Res}_H^G V \right)
            =
            \left(\chi ,\operatorname{Res}_H^G \psi \right)_H
        \end{equation*}
    \end{proof}
\end{theorem}

\section{Mackey's Decomposition and irreducibility criterion}

Let $G=\bigsqcup_{g\in K\backslash G/H} KgH$ be a double coset decomposition of $G$.

\begin{proposition}
    \begin{equation*}
        k[G]
        \cong
        \bigoplus_{g\in K\backslash G/H} k[KgH]
    \end{equation*}
    as $k[K]$-$k[H]$-bimodules.
\end{proposition}

\begin{proposition}
    For a fixed $g \in G$, we have
    \begin{equation*}
        k[KgH]
        \cong
        k[K] \otimes_{k[K\cap gHg^{-1}]} k[gH]
    \end{equation*}
    as $k[K]$-$k[H]$-bimodules.
\end{proposition}


\begin{theorem}[Mackey's Decomposition Theorem]
    Then
    \begin{equation*}
        \operatorname{Res}_K^G \circ \operatorname{Ind}_H^G
        \cong
        \bigoplus_{g\in K\backslash G/H} \operatorname{Ind}_{K\cap gHg^{-1}}^K \circ {}^g \operatorname{Res}^{H}_{H\cap gKg^{-1}}
    \end{equation*}
    where ${}^g W$ is the conjugate module of $W$ by $g$.
    \begin{proof}
        Given representation $W$ of $H$, we have
        \begin{equation*}
            \begin{aligned}
                \operatorname{Res}_K^G \circ \operatorname{Ind}_H^G \left(W\right) &
                = k[K]\otimes_{k[K]} k[G] \otimes_{k[H]} W                                                                                                                                             \\
                                                                                   & \cong \bigoplus_{g\in K\backslash G/H} k[K]\otimes_{k[K]} k[K] \otimes_{k[K\cap gHg^{-1}]} k[gH] \otimes_{k[H]} W \\
                                                                                   & \cong \bigoplus_{g\in K\backslash G/H} k[K] \otimes_{k[K\cap gHg^{-1}]} k[gH] \otimes_{k[H]} W                    \\
            \end{aligned}
        \end{equation*}

        Noted that $k[gH]\otimes_{k[H]} W $ is $k[K\cap gHg^{-1}]$-module which is isomorphic to ${}^g \operatorname{Res}^{H}_{H\cap gKg^{-1}} W$ as $k[K\cap gHg^{-1}]$-module.
    \end{proof}
\end{theorem}
