\section{Matrix Multiplication and Determinants}

\begin{question}
    Compute the determinant of the following circulant matrix $\mathbf{A}$:
    \[
        \mathbf{A}
        =
        \left(\begin{array}{ccccc}
                a_1     & a_2    & a_3    & \cdots & a_n     \\
                a_n     & a_1    & a_2    & \cdots & a_{n-1} \\
                a_{n-1} & a_n    & a_1    & \cdots & a_{n-2} \\
                \vdots  & \vdots & \vdots &        & \vdots  \\
                a_2     & a_3    & a_4    & \cdots & a_1
            \end{array}\right)
    \]
    \begin{proof}
        Define the polynomial $f(x)=a_1+a_2 x+a_3 x^2+\cdots+a_n x^{n-1}$, and let $\varepsilon_1, \varepsilon_2, \cdots, \varepsilon_n$ be all the $n$th roots of unity. Also, let
        \[
            \mathbf{V}
            =
            \left(\begin{array}{ccccc}
                    1                   & 1                   & 1                   & \cdots & 1                   \\
                    \varepsilon_1       & \varepsilon_2       & \varepsilon_3       & \cdots & \varepsilon_n       \\
                    \varepsilon_1^2     & \varepsilon_2^2     & \varepsilon_3^2     & \cdots & \varepsilon_n^2     \\
                    \vdots              & \vdots              & \vdots              &        & \vdots              \\
                    \varepsilon_1^{n-1} & \varepsilon_2^{n-1} & \varepsilon_3^{n-1} & \cdots & \varepsilon_n^{n-1}
                \end{array}
            \right)
        \]
        Then
        \[
            \mathbf{A} \mathbf{V}
            =
            \left(\begin{array}{ccccc}
                    f\left(\varepsilon_1\right)                     & f\left(\varepsilon_2\right)                     & f\left(\varepsilon_3\right)                     & \cdots & f\left(\varepsilon_n\right)                     \\
                    \varepsilon_1 f\left(\varepsilon_1\right)       & \varepsilon_2 f\left(\varepsilon_2\right)       & \varepsilon_3 f\left(\varepsilon_3\right)       & \cdots & \varepsilon_n f\left(\varepsilon_n\right)       \\
                    \varepsilon_1^2 f\left(\varepsilon_1\right)     & \varepsilon_2^2 f\left(\varepsilon_2\right)     & \varepsilon_3^2 f\left(\varepsilon_3\right)     & \cdots & \varepsilon_n^2 f\left(\varepsilon_n\right)     \\
                    \vdots                                          & \vdots                                          & \vdots                                          &        & \vdots                                          \\
                    \varepsilon_1^{n-1} f\left(\varepsilon_1\right) & \varepsilon_2^{n-1} f\left(\varepsilon_2\right) & \varepsilon_3^{n-1} f\left(\varepsilon_3\right) & \cdots & \varepsilon_n^{n-1} f\left(\varepsilon_n\right)
                \end{array}\right)
        \]
        Therefore,
        \[
            |\mathbf{A}||\mathbf{V}|=|\mathbf{A} \mathbf{V}|
            =f(\varepsilon_1)f(\varepsilon_2)\cdots f(\varepsilon_n)|\mathbf{V}|
        \]
        Since the $\varepsilon_i$ are pairwise distinct, $|\mathbf{V}| \neq 0$, and hence
        \[
            |\mathbf{A}|
            =
            f\left(\varepsilon_1\right) f\left(\varepsilon_2\right) \cdots f\left(\varepsilon_n\right)
        \]
    \end{proof}
\end{question}
\begin{question}
    Compute the determinant of the following matrix $\mathbf{A}$:
    \[
        \mathbf{A}
        =
        \left(
        \begin{array}{cccc}
                x & -y & -z & -w \\
                y & x  & -w & z  \\
                z & w  & x  & -y \\
                w & -z & y  & x
            \end{array}
        \right)
    \]
    \begin{proof}[1st Proof]
        Observe that
        \[
            \mathbf{A} \mathbf{A}^{\prime}=\left(\begin{array}{cccc}
                    x & -y & -z & -w \\
                    y & x  & -w & z  \\
                    z & w  & x  & -y \\
                    w & -z & y  & x
                \end{array}\right)\left(\begin{array}{cccc}
                    x  & y  & z  & w  \\
                    -y & x  & w  & -z \\
                    -z & -w & x  & y  \\
                    -w & z  & -y & x
                \end{array}\right)=\left(\begin{array}{cccc}
                    u & 0 & 0 & 0 \\
                    0 & u & 0 & 0 \\
                    0 & 0 & u & 0 \\
                    0 & 0 & 0 & u
                \end{array}\right),
        \]
        where $u=x^2+y^2+z^2+w^2$. Therefore,
        \[
            |\mathbf{A}|^2=\left(x^2+y^2+z^2+w^2\right)^4 .
        \]
        Setting $x=1$ and $y=z=w=0$ in $\mathbf{A}$ gives $|\mathbf{A}|=1$, so
        \[
            |\mathbf{A}|=\left(x^2+y^2+z^2+w^2\right)^2 .
        \]


    \end{proof}
    \begin{proof}[2nd Proof]
        Let
        \[
            \mathbf{B}=\left(\begin{array}{cc}
                    x & -y \\
                    y & x
                \end{array}\right), \quad \mathbf{C}=\left(\begin{array}{cc}
                    z & w  \\
                    w & -z
                \end{array}\right)
        \]
        Then $|\mathbf{A}|=\left|\begin{array}{cc}\mathbf{B} & -\mathbf{C} \\ \mathbf{C} & \mathbf{B}\end{array}\right|$,
        \[
            \begin{aligned}
                |\mathbf{A}| & =|\mathbf{B}+\mathrm{i} \mathbf{C}||\mathbf{B}-\mathrm{i} \mathbf{C}| \\
                             & =\left|\begin{array}{cc}
                                          x+\mathrm{i} z & -y+\mathrm{i} w \\
                                          y+\mathrm{i} w & x-\mathrm{i} z
                                      \end{array}\right|\left|\begin{array}{cc}
                                                                  x-\mathrm{i} z & -y-\mathrm{i} w \\
                                                                  y-\mathrm{i} w & x+\mathrm{i} z
                                                              \end{array}\right|       \\
                             & =\left(x^2+y^2+z^2+w^2\right)^2
            \end{aligned}
        \]
    \end{proof}
\end{question}




\section{The Cauchy--Binet Formula}
\begin{theorem}
    Let $\mathbf{A}=\left(a_{i j}\right)$ be an $m \times n$ matrix, let $\mathbf{B}=\left(b_{i j}\right)$ be an $n \times m$ matrix, and let $r$ be a positive integer with $r \leq m$. Then every minor of order $r$ of ${\mathbf{A} \mathbf{B}}$ satisfies
    \[
        \mathbf{A} \mathbf{B}
        \left(
        \begin{matrix}
                i_1 & i_2 & \cdots & i_r \\
                j_1 & j_2 & \cdots & j_r
            \end{matrix}
        \right)
        =
        \sum_{1 \leq k_1<\cdots<k_r \leq n} \mathbf{A}
        \left(
        \begin{matrix}
                i_1 & i_2 & \cdots & i_r \\
                k_1 & k_2 & \cdots & k_r
            \end{matrix}
        \right)
        \mathbf{B}
        \left(
        \begin{matrix}
                k_1 & k_2 & \cdots & k_r \\
                j_1 & j_2 & \cdots & j_r
            \end{matrix}
        \right)
    \]
\end{theorem}


\begin{theorem}
    Let $\mathbf{A}$ and $\mathbf{B}$ be $m \times n$ and $n \times m$ matrices, respectively. Prove that the sums of the principal minors of order $r$ of $\mathbf{A} \mathbf{B}$ and $\mathbf{B} \mathbf{A}$ (the two quantities $\sigma_r$) are equal, where $1 \leq r \leq \min \{m, n\}$.
    \begin{proof}
        By the Cauchy--Binet formula,
        \[
            \begin{aligned}
                  & \sum_{1 \leq i_1<i_2<\cdots<i_r \leq m} {\mathbf{A} \mathbf{B}}\left(\begin{array}{llll}
                                                                                                 i_1 & i_2 & \cdots & i_r \\
                                                                                                 i_1 & i_2 & \cdots & i_r
                                                                                             \end{array}\right)                                                               \\
                = & \sum_{1 \leq i_1<i_2<\cdots<i_r \leq m} \sum_{1 \leq j_1<j_2<\cdots<j_r \leq n} \mathbf{A}\left(\begin{array}{llll}
                                                                                                                            i_1 & i_2 & \cdots & i_r \\
                                                                                                                            j_1 & j_2 & \cdots & j_r
                                                                                                                        \end{array}\right) \mathbf{B}\left(\begin{array}{llll}
                                                                                                                                                               j_1 & j_2 & \cdots & j_r \\
                                                                                                                                                               i_1 & i_2 & \cdots & i_r
                                                                                                                                                           \end{array}\right) \\
                = & \sum_{1 \leq j_1<j_2<\cdots<j_r \leq n} \sum_{1 \leq i_1<i_2<\cdots<i_r \leq m} \mathbf{B}\left(\begin{array}{lllll}
                                                                                                                            j_1 & j_2 & \cdots & j_r \\
                                                                                                                            i_1 & i_2 & \cdots & i_r
                                                                                                                        \end{array}\right) \mathbf{A}\left(\begin{array}{llll}
                                                                                                                                                               i_1 & i_2 & \cdots & i_r \\
                                                                                                                                                               j_1 & j_2 & \cdots & j_r
                                                                                                                                                           \end{array}\right) \\
                = & \sum_{1 \leq j_1<j_2<\cdots<j_r \leq n} {\mathbf{B} \mathbf{A}}\left(\begin{array}{llll}
                                                                                                 j_1 & j_2 & \cdots & j_r \\
                                                                                                 j_1 & j_2 & \cdots & j_r
                                                                                             \end{array}\right)
            \end{aligned}
        \]
    \end{proof}
\end{theorem}

\begin{corollary}
    $\mathbf{A}\mathbf{B}$ and $\mathbf{B}\mathbf{A}$ have the same nonzero eigenvalues, counted with multiplicity.
\end{corollary}

\section{Elementary Block Transformations and Reduction Formulas}
\begin{theorem}[Determinant Reduction Formula]
    Let $\mathbf{A}$ be an $m \times m$ matrix, $\mathbf{D}$ an $n \times n$ matrix, $\mathbf{B}$ an $m \times n$ matrix, and $\mathbf{C}$ an $n \times m$ matrix. Prove the following:

    (1) If $\mathbf{A}$ is invertible, then
    \[
        \left|
        \begin{matrix}
            \mathbf{A} & \mathbf{B} \\
            \mathbf{C} & \mathbf{D}
        \end{matrix}
        \right|
        =
        \left|\mathbf{A}\right|\left|\mathbf{D}-\mathbf{C} \mathbf{A}^{-1} \mathbf{B}\right|
    \]

    (2) If $\mathbf{D}$ is invertible, then
    \[
        \left|
        \begin{matrix}
            \mathbf{A} & \mathbf{B} \\
            \mathbf{C} & \mathbf{D}
        \end{matrix}
        \right|
        =
        \left|\mathbf{D} \| \mathbf{A}-\mathbf{B} \mathbf{D}^{-1} \mathbf{C}\right| ;
    \]

    (3) If both $\mathbf{A}$ and $\mathbf{D}$ are invertible, then
    \[
        \left|\mathbf{D} \| \mathbf{A}-\mathbf{B} \mathbf{D}^{-1} \mathbf{C}\right|=|\mathbf{A}|\left|\mathbf{D}-\mathbf{C} \mathbf{A}^{-1} \mathbf{B}\right|
    \]
\end{theorem}
\begin{corollary}
    Let $\mathbf{A}$ be an $n\times m$ matrix and $\mathbf{B}$ an $m\times n$ matrix. Then
    \[
        \left|\lambda \mathbf{I}_n -\mathbf{A}\mathbf{B}\right|
        =
        \lambda^{n-m}\left|\lambda \mathbf{I}_m -\mathbf{B}\mathbf{A}\right|
    \]
\end{corollary}

\begin{theorem}
    Let $\mathbf{A}$ and $\mathbf{B}$ be $n \times n$ matrices. Prove that
    \[
        \left|
        \begin{matrix}
            \mathbf{A} & \mathbf{B} \\
            \mathbf{B} & \mathbf{A}
        \end{matrix}
        \right|
        =
        |\mathbf{A}+\mathbf{B}||\mathbf{A}-\mathbf{B}|
    \]
    \[
        \left|
        \begin{matrix}
            \mathbf{A} & -\mathbf{B} \\
            \mathbf{B} & \mathbf{A}
        \end{matrix}
        \right|
        =
        |\mathbf{A}+\mathrm{i} \mathbf{B} \| \mathbf{A}-\mathrm{i} \mathbf{B}|
    \]
\end{theorem}
\begin{corollary}
    If $\mathbf{A}$ and $\mathbf{B}$ are real matrices, then
    \[
        \left|
        \begin{matrix}
            \mathbf{A} & -\mathbf{B} \\
            \mathbf{B} & \mathbf{A}
        \end{matrix}
        \right|
        =
        |\mathbf{A}+\mathrm{i} \mathbf{B} \| \mathbf{A}-\mathrm{i} \mathbf{B}|
        =
        \left|\mathrm{det}(\mathbf{A}+\mathrm{i} \mathbf{B})\right|^2 \geq 0
    \]
\end{corollary}

\begin{theorem}[Matrix Units and Standard Unit Vectors]
    The standard unit vectors are the following $n$ column vectors in dimension $n$:
    \[
        {e}_1=\left(\begin{array}{c}
                1      \\
                0      \\
                \vdots \\
                0
            \end{array}\right), \quad {e}_2=\left(\begin{array}{c}
                0      \\
                1      \\
                \vdots \\
                0
            \end{array}\right), \cdots, \quad {e}_n=\left(\begin{array}{c}
                0      \\
                0      \\
                \vdots \\
                1
            \end{array}\right)
    \]
    The vectors ${e}_1^{\prime}, {e}_2^{\prime}, \cdots, {e}_n^{\prime}$ are called the standard unit row vectors in dimension $n$. Let $f_1, f_2, \cdots, f_m$ be the standard unit column vectors in dimension $m$.

    (1)
    \[
        {e}_i^{\prime} {e}_j
        =
        \delta_{ij}
    \]

    \[
        {e}_i{e}_j^{\prime}
        =
        \mathbf{E}_{i j}
    \]


    (2) If $\mathbf{A}=\left(a_{i j}\right)$ is an $m \times n$ matrix, then $\mathbf{A} {e}_i$ is the $i$th column of $\mathbf{A}$:
    \[
        \left({\alpha_1},{\alpha_2},\ldots,{\alpha_n}\right) {e}_i
        =
        {\alpha}_i
    \]
    Likewise, ${f}_i^{\prime} \mathbf{A}$ is the $i$th row of $\mathbf{A}$:
    \[
        {f}_i^\prime
        \left(
        \begin{matrix}
                {\beta}_1 \\
                {\beta}_2 \\
                \vdots    \\
                {\beta}_m
            \end{matrix}
        \right)
        =
        {\beta}_i
    \]

    (3) If $\mathbf{A}=\left(a_{i j}\right)$ is an $m \times n$ matrix, then ${f}_i^{\prime} \mathbf{A} {e}_j=a_{i j}$.

    (4)



    \[
        \mathbf{E}_{i j} \mathbf{E}_{k l}
        =
        \begin{cases}
            \mathbf{E}_{i l} & ,j=k     \\
            0                & ,j\neq k
        \end{cases}
    \]

    (5) If $\mathbf{A}$ is an $n \times n$ matrix and $\mathbf{A}=\left(a_{i j}\right)$, then the $i$th row of $\mathbf{E}_{i j} \mathbf{A}$ is the $j$th row of $\mathbf{A}$, while all other rows of $\mathbf{E}_{i j} \mathbf{A}$ are zero.

    If $\mathbf{A}$ is an $n \times n$ matrix and $\mathbf{A}=\left(a_{i j}\right)$, then the $j$th column of $\mathbf{A} \mathbf{E}_{i j}$ is the $i$th column of $\mathbf{A}$, while all other columns of $\mathbf{A} \mathbf{E}_{i j}$ are zero.

    (6) If $\mathbf{A}$ is an $n \times n$ matrix and $\mathbf{A}=\left(a_{i j}\right)$, then $\mathbf{E}_{i j} \mathbf{A} \mathbf{E}_{k l}=a_{j k} \mathbf{E}_{i l}$.

\end{theorem}










\begin{theorem}[Basic Cyclic Matrix]
    Let the basic cyclic matrix of order $n$ be
    \[
        \mathbf{C}
        =
        \left(
        \begin{matrix}
                0      & 1      & 0      & \cdots & 0      \\
                0      & 0      & 1      & \cdots & 0      \\
                \vdots & \vdots & \vdots &        & \vdots \\
                0      & 0      & 0      & \cdots & 1      \\
                1      & 0      & 0      & \cdots & 0
            \end{matrix}
        \right)
    \]

    (1) Circulant matrices are closed under polynomial evaluation, transposition, and inversion (when the inverse exists).

    (2) A matrix is circulant if and only if it can be written as $f(\mathbf{C})$ for some $f \in \mathbb{F}[x]$.

    (3) The equality $\mathbf{A}\mathbf{C}=\mathbf{C}\mathbf{A}$ holds if and only if $\mathbf{A}$ is circulant.
\end{theorem}

Their diagonal entries are obtained by applying the corresponding operation to the original diagonal entries.
\end{theorem}




\section{Invertible Matrices}
\begin{theorem}
    Let $\mathbf{A}$ be an $m \times n$ matrix and $\mathbf{B}$ an $n \times m$ matrix. If $\mathbf{I}_m+{\mathbf{A} \mathbf{B}}$ is invertible, then $\mathbf{I}_n+\mathbf{B} \mathbf{A}$ is also invertible.
    \[
        \left(\mathbf{I}_n+\mathbf{B} \mathbf{A}\right)^{-1}=\mathbf{I}_n-\mathbf{B}\left(\mathbf{I}_m+\mathbf{A} \mathbf{B}\right)^{-1} \mathbf{A}
    \]

    \textbf{Proof.}
    Observe that $\mathbf{A}\left(\mathbf{I}_n+\mathbf{B} \mathbf{A}\right)=\left(\mathbf{I}_m+\mathbf{A} \mathbf{B}\right) \mathbf{A}$. Hence $\left(\mathbf{I}_m+\mathbf{A} \mathbf{B}\right)^{-1} \mathbf{A}\left(\mathbf{I}_n+\mathbf{B} \mathbf{A}\right)=\mathbf{A}$, and therefore $\mathbf{B}\left(\mathbf{I}_m+\mathbf{A} \mathbf{B}\right)^{-1} \mathbf{A}\left(\mathbf{I}_n+\mathbf{B} \mathbf{A}\right)=\mathbf{B} \mathbf{A}$. It follows that

    \[
        \begin{aligned}
            \mathbf{I}_n & =\mathbf{I}_n+{\mathbf{B} \mathbf{A}}-\mathbf{B} \mathbf{A}=\left(\mathbf{I}_n+{\mathbf{B} \mathbf{A}}\right)-\mathbf{B}\left(\mathbf{I}_m+{\mathbf{A} \mathbf{B}}\right)^{-1} \mathbf{A}\left(\mathbf{I}_n+{\mathbf{B} \mathbf{A}}\right) \\
                         & =\left(\mathbf{I}_n-\mathbf{B}\left(\mathbf{I}_m+{\mathbf{A} \mathbf{B}}\right)^{-1} \mathbf{A}\right)\left(\mathbf{I}_n+{\mathbf{B} \mathbf{A}}\right) .
        \end{aligned}
    \]
    Thus $\left(\mathbf{I}_n+\mathbf{B} \mathbf{A}\right)^{-1}=\mathbf{I}_n-\mathbf{B}\left(\mathbf{I}_m+\mathbf{A} \mathbf{B}\right)^{-1} \mathbf{A}$.
\end{theorem}

\begin{corollary}
    Let $\mathbf{A}$ and $\mathbf{B}$ be invertible $n \times n$ matrices, and suppose that $\mathbf{A}^{-1}+\mathbf{B}^{-1}$ is invertible. Prove that $\mathbf{A}+\mathbf{B}$ is also invertible and that
    \[
        (\mathbf{A}+\mathbf{B})^{-1}=\mathbf{A}^{-1}-\mathbf{A}^{-1}\left(\mathbf{A}^{-1}+\mathbf{B}^{-1}\right)^{-1} \mathbf{A}^{-1}
    \]
\end{corollary}


\begin{corollary}[Sherman--Morrison Formula]
    Let $\mathbf{A}$ be an invertible $n \times n$ matrix, and let ${\alpha}$ and ${\beta}$ be $n$-dimensional column vectors such that $1+{\beta}^{\prime} \mathbf{A}^{-1} {\alpha} \neq 0$. Prove that
    \[
        \left(\mathbf{A}+{\alpha} {\beta}^{\prime}\right)^{-1}=\mathbf{A}^{-1}-\frac{1}{1+{\beta}^{\prime} \mathbf{A}^{-1} {\alpha}} \mathbf{A}^{-1} {\alpha} {\beta}^{\prime} \mathbf{A}^{-1}
    \]
\end{corollary}

\begin{theorem}
    Let $\mathbf{A}$ and $\mathbf{B}$ be $n \times n$ matrices and let $k\in \mathbb{F}$. Then

    (1) $\operatorname{tr}(\mathbf{A}+\mathbf{B})=\operatorname{tr}(\mathbf{A})+\operatorname{tr}(\mathbf{B})$

    (2) $\operatorname{tr}(k \mathbf{A})=k \operatorname{tr}(\mathbf{A})$

    (3) $\operatorname{tr}\left(\mathbf{A}^{\prime}\right)=\operatorname{tr}(\mathbf{A})$

    (4) If $\mathbf{A}$ is an $m\times n$ matrix and $\mathbf{B}$ is an $n\times m$ matrix, then $\operatorname{tr}({\mathbf{A} \mathbf{B}})=\operatorname{tr}(\mathbf{B} \mathbf{A})$.

    (5) $\mathrm{tr}\left(\mathbf{P}^{-1}\mathbf{A}\mathbf{P}\right) = \mathrm{tr}\left(\mathbf{A}\right)$
\end{theorem}


\begin{question}
    Let $\mathbf{A}=\left(a_{i j}\right)$ be an $n \times n$ matrix and define $f(\mathbf{A})=\sum_{i, j=1}^n a_{i j}^2$. Suppose $\mathbf{P}$ is an invertible $n \times n$ matrix such that $f\left(\mathbf{P} \mathbf{A} \mathbf{P}^{-1}\right)=f(\mathbf{A})$ for every $n \times n$ matrix $\mathbf{A}$. Prove that there exists a nonzero constant $c$ such that $\mathbf{P}^{\prime} \mathbf{P}=c \mathbf{I}_n$.

    \textbf{Proof.}
    We compute that
    \[
        \begin{aligned}
            f(\mathbf{P}\mathbf{A}\mathbf{P}^{-1}) & =
            \mathrm{tr}(\mathbf{P}\mathbf{A}\mathbf{P}^{-1})                                                                                                        \\
                                                   & = \mathrm{tr}\left(\mathbf{P}\mathbf{A}\mathbf{P}^{-1}\left(\mathbf{P}\mathbf{A}\mathbf{P}^{-1}\right)'\right) \\
                                                   & = \mathrm{tr}\left(\mathbf{U} \mathbf{A} \mathbf{U}^{-1}\mathbf{A}'\right)                                     \\
        \end{aligned}
    \]
    where $\mathbf{U}=\mathbf{P}'\mathbf{P}$. We set
    \[
        \mathbf{A}=\sum_{i,j}a_{ij} \mathbf{E}_{ij}
    \]
    Then
    \[
        \begin{aligned}
            \mathrm{tr}\left(\mathbf{U} \mathbf{A} \mathbf{U}^{-1}\mathbf{A}'\right) & =\mathrm{tr}\left(\mathbf{U} \left(\sum_{i,j}a_{ij} \mathbf{E}_{ij}\right) \mathbf{U}^{-1} \left(\sum_{k,l}a_{kl} \mathbf{E}_{kl}\right)\right) \\
                                                                                     & =\mathrm{tr}\left(\sum_{i,j,k,l} a_{ij}a_{kl} \mathbf{U} e_ie_j' \mathbf{U}^{-1} e_ke_l'\right)                                                 \\
                                                                                     & =\mathrm{tr}\left(\sum_{i,j,k,l} a_{ij}a_{kl} e_j' \mathbf{U}^{-1} e_ke_l'\mathbf{U} e_i\right)                                                 \\
                                                                                     & =\sum_{i,j,k,l} a_{ij}a_{kl}\left( e_j' \mathbf{U}^{-1} e_k\right) \left(e_l'\mathbf{U} e_i\right)
        \end{aligned}
    \]
    and
    \[
        \mathrm{tr}(\mathbf{A}'\mathbf{A})=\sum_{m,n} a_{mn}^2
    \]
    Therefore, we have
    \[
        \sum_{i,j,k,l} a_{ij}a_{kl}\left( e_j' \mathbf{U}^{-1} e_k\right) \left(e_l'\mathbf{U} e_i\right)
        =
        \sum_{m,n} a_{mn}^2
    \]
    for all $\left(a_{ij}\right)_{n\times n}$. We conclude that $\mathbf{U}=c\mathbf{I}_n$.
\end{question}

\begin{theorem}
    Basic formulas for matrix rank:

    (1) If $k \neq 0$, then $\mathrm{r}(k \mathbf{A})=\mathrm{r}(\mathbf{A})$.

    (1) $\mathrm{r}(\mathbf{A} \mathbf{B}) \leq \min \{\mathrm{r}(\mathbf{A}), \mathrm{r}(\mathbf{B})\}$

    (2) $\mathrm{r}
        \left(
        \begin{matrix}
                \mathbf{A} & \mathbf{O} \\ \mathbf{O} & \mathbf{B}
            \end{matrix}
        \right)
        =
        \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})$

    (3) $\mathrm{r}\left(
        \begin{matrix}
                \mathbf{A} & \mathbf{O} \\ \mathbf{O} & \mathbf{B}
            \end{matrix}
        \right)
        \geq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})
        ,
        \mathrm{r}
        \left(
        \begin{matrix}\mathbf{A} & \mathbf{O} \\ \mathbf{D} & \mathbf{B}
            \end{matrix}
        \right)
        \geq
        \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})$

    (4) $\mathrm{r}(\mathbf{A} ; \mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B}), \mathrm{r}\binom{\mathbf{A}}{\mathbf{B}} \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})$

    (5) $\mathrm{r}(\mathbf{A}+\mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B}), \mathrm{r}(\mathbf{A}-\mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})$

    (6) $\mathrm{r}(\mathbf{A}-\mathbf{B}) \geq|\mathrm{r}(\mathbf{A})-\mathrm{r}(\mathbf{B})|$
\end{theorem}


\begin{theorem}[Frobenius Inequality]
    Prove that $\mathrm{r}(\mathbf{A} \mathbf{B} \mathbf{C}) \geq \mathrm{r}({\mathbf{A} \mathbf{B}})+\mathrm{r}(\mathbf{B} \mathbf{C})-\mathrm{r}(\mathbf{B})$.

    \textbf{Proof.} Consider the following elementary block transformations:
    \[
        \left(\begin{array}{cc}
                \mathbf{A} \mathbf{B} \mathbf{C} & \mathbf{O} \\
                \mathbf{O}                       & \mathbf{B}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{A} \mathbf{B} \mathbf{C} & \mathbf{A} \mathbf{B} \\
                \mathbf{O}                       & \mathbf{B}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{O}             & \mathbf{A} \mathbf{B} \\
                -\mathbf{B} \mathbf{C} & \mathbf{B}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{A} \mathbf{B} & \mathbf{O}            \\
                \mathbf{B}            & \mathbf{B} \mathbf{C}
            \end{array}\right)
    \]
    which give
    \[
        \mathrm{r}(\mathbf{A} \mathbf{B} \mathbf{C})+\mathrm{r}(\mathbf{B})=\mathrm{r}\left(\begin{array}{cc}
                \mathbf{A} \mathbf{B} \mathbf{C} & \mathbf{O} \\
                \mathbf{O}                       & \mathbf{B}
            \end{array}\right)=\mathrm{r}\left(\begin{array}{cc}
                \mathbf{A} \mathbf{B} & \mathbf{O}            \\
                \mathbf{B}            & \mathbf{B} \mathbf{C}
            \end{array}\right) \geq \mathrm{r}(\mathbf{A} \mathbf{B})+\mathrm{r}({\mathbf{B} \mathbf{C}})
    \]
    The result follows.
\end{theorem}
\begin{corollary}[Sylvester Inequality]
    Let $\mathbf{A}$ be an $m \times n$ matrix and $\mathbf{B}$ an $n \times t$ matrix. Prove that
    \[
        \mathrm{r}(\mathbf{A} \mathbf{B}) \geq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})-n
    \]

    \textbf{Proof.} Consider the following elementary block transformations:
    \[
        \left(\begin{array}{cc}
                \mathbf{I}_n & \mathbf{O}            \\
                \mathbf{O}   & \mathbf{A} \mathbf{B}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{I}_n & \mathbf{O}            \\
                \mathbf{A}   & \mathbf{A} \mathbf{B}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{I}_n & -\mathbf{B} \\
                \mathbf{A}   & \mathbf{O}
            \end{array}\right) \rightarrow\left(\begin{array}{cc}
                \mathbf{B} & \mathbf{I}_n \\
                \mathbf{O} & \mathbf{A}
            \end{array}\right)
    \]
    which give
    \[
        \mathrm{r}(\mathbf{A} \mathbf{B})+n=\mathrm{r}\left(\begin{array}{cc}
                \mathbf{I}_n & \mathbf{O}            \\
                \mathbf{O}   & \mathbf{A} \mathbf{B}
            \end{array}\right)=\mathrm{r}\left(\begin{array}{cc}
                \mathbf{B} & \mathbf{I}_n \\
                \mathbf{O} & \mathbf{A}
            \end{array}\right) \geq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})
    \]
    Thus $\mathrm{r}(\mathbf{A} \mathbf{B}) \geq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})-n$.
\end{corollary}

\begin{corollary}
    If $\mathbf{A}$ is an $m \times n$ matrix, $\mathbf{B}$ is an $n \times t$ matrix, and ${\mathbf{A} \mathbf{B}}=\mathbf{O}$, then $\mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B}) \leq n$.
\end{corollary}

\begin{corollary}
    Let $\mathbf{A}$ be an $n \times n$ matrix. Then

        0 \geq r(\mathbf{A}^{n+2})- r(\mathbf{A}^{n+1})
        \geq r(\mathbf{A}^{n+1}) -r(\mathbf{A}^n)
        \geq \cdots \geq r(\mathbf{A})-n
    \]
\end{corollary}


\begin{theorem}[Rank Reduction Formula]
    Let $\mathbf{M}=\left(\begin{matrix}\mathbf{A} & \mathbf{B} \\ \mathbf{C} & \mathbf{D}\end{matrix}\right)$ be a block matrix. Prove the following:

    (1) If $\mathbf{A}$ is invertible, then $\mathrm{r}(\mathbf{M})=\mathrm{r}(\mathbf{A})+\mathrm{r}\left(\mathbf{D}-{\mathbf{C} \mathbf{A}}^{-1} \mathbf{B}\right)$.

\end{theorem}

\begin{question}
    Let $\mathbf{A}$ and $\mathbf{B}$ be $n \times n$ matrices over a field $\mathbb{K}$ such that $\mathbf{A} \mathbf{B}=\mathbf{B} \mathbf{A}$. Prove that
    \[
        \mathrm{r}(\mathbf{A}+\mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})-\mathrm{r}(\mathbf{A} \mathbf{B})
    \]

    \textbf{Proof 1 (elementary matrix transformations).}
    Consider the following multiplication of block matrices:
    \[
        \left(\begin{matrix}
                \mathbf{I} & \mathbf{I} \\
                \mathbf{O} & \mathbf{I}
            \end{matrix}\right)\left(\begin{matrix}
                \mathbf{A} & \mathbf{O} \\
                \mathbf{O} & \mathbf{B}
            \end{matrix}\right)\left(\begin{matrix}
                \mathbf{I} & -\mathbf{B} \\
                \mathbf{I} & \mathbf{A}
            \end{matrix}\right)=\left(\begin{matrix}
                \mathbf{A}+\mathbf{B} & -\mathbf{A} \mathbf{B}+\mathbf{B} \mathbf{A} \\
                \mathbf{B}            & \mathbf{B} \mathbf{A}
            \end{matrix}\right)=\left(\begin{matrix}
                \mathbf{A}+\mathbf{B} & \mathbf{O}            \\
                \mathbf{B}            & \mathbf{A} \mathbf{B}
            \end{matrix}\right)
    \]
    which gives
    \[
        \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})
        =
        \mathrm{r}\left(\begin{matrix}
                \mathbf{A} & \mathbf{O} \\
                \mathbf{O} & \mathbf{B}
            \end{matrix}\right)
        \geq
        \mathrm{r}\left(\begin{matrix}
                \mathbf{A}+\mathbf{B} & \mathbf{O}            \\
                \mathbf{B}            & \mathbf{B} \mathbf{A}
            \end{matrix}\right)
        \geq
        \mathrm{r}(\mathbf{A}+\mathbf{B})+\mathrm{r}(\mathbf{A} \mathbf{B}),
    \]
    The result follows.

    \textbf{Proof 2 (the kernel-space approach).}
    Let $V_\mathbf{A}$ be the solution space of ${\mathbf{A} x}=0$, and define $V_\mathbf{B}, V_{\mathbf{A} \mathbf{B}}, V_{\mathbf{A}+\mathbf{B}}$ analogously. We have $V_\mathbf{A} \cap V_\mathbf{B} \subseteq V_{\mathbf{A}+\mathbf{B}}$, $V_\mathbf{A} \subseteq V_{\mathbf{B} \mathbf{A}}$, and $V_\mathbf{B} \subseteq V_{\mathbf{A} \mathbf{B}}$. Since ${\mathbf{A} \mathbf{B}}=\mathbf{B} \mathbf{A}$, we have $V_{\mathbf{B} \mathbf{A}}=V_{\mathbf{A} \mathbf{B}}$, and hence $V_\mathbf{A}+V_\mathbf{B} \subseteq V_{\mathbf{A} \mathbf{B}}$. Therefore,
    \[
        \operatorname{dim}\left(V_{\mathbf{A}} \cap V_{\mathbf{B}}\right) \leq \operatorname{dim} V_{\mathbf{A}+\mathbf{B}}=n-\mathrm{r}(\mathbf{A}+\mathbf{B})
    \]
    and
    \[
        \operatorname{dim}\left(V_{\mathbf{A}}+V_{\mathbf{B}}\right) \leq \operatorname{dim} V_{\mathbf{A} \mathbf{B}}=n-\mathrm{r}(\mathbf{A} \mathbf{B})
    \]
    Adding these inequalities and using the dimension formula for sums and intersections gives
    \[
        \begin{aligned}
            n-\mathrm{r}(\mathbf{A}+\mathbf{B})+n-\mathrm{r}(\mathbf{A} \mathbf{B}) & \geq \operatorname{dim}\left(V_{\mathbf{A}} \cap V_{\mathbf{B}}\right)+\operatorname{dim}\left(V_{\mathbf{A}}+V_{\mathbf{B}}\right) \\
                                                                                    & =\operatorname{dim} V_{\mathbf{A}}+\operatorname{dim} V_{\mathbf{B}}                                                                \\
                                                                                    & =n-\mathrm{r}(\mathbf{A})+n-\mathrm{r}(\mathbf{B})
        \end{aligned}
    \]
    Therefore, $\mathrm{r}(\mathbf{A}+\mathbf{B})+\mathrm{r}(\mathbf{A} \mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})$, as required.

    \[
        \mathrm{r}(\mathbf{A}+\mathbf{B}) \leq \mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B})-\mathrm{r}(\mathbf{A} \mathbf{B})
    \]

    \textbf{Proof 3 (the image-space approach).}
    Write the column decompositions $\mathbf{A}=\left({\alpha}_1, {\alpha}_2, \cdots, {\alpha}_n\right)$ and $\mathbf{B}=\left({\beta}_1, {\beta}_2, \cdots, {\beta}_n\right)$. Let $U_{\mathbf{A}}=L\left({\alpha}_1, {\alpha}_2, \cdots, {\alpha}_n\right)$ be the subspace of $\mathbb{K}^n$ spanned by the columns of $\mathbf{A}$, and define $U_{\mathbf{B}}, U_{{\mathbf{A} \mathbf{B}}}, U_{\mathbf{A}+\mathbf{B}}$ analogously.

    Clearly, $U_{\mathbf{A}+\mathbf{B}} \subseteq U_{\mathbf{A}}+U_{\mathbf{B}}$. Since ${\mathbf{A} \mathbf{B}}=\left(\mathbf{A} {\beta}_1, \mathbf{A} {\beta}_2, \cdots, \mathbf{A} {\beta}_n\right)$, we have $U_{\mathbf{A} \mathbf{B}} \subseteq U_{\mathbf{A}}$. Moreover, ${\mathbf{A} \mathbf{B}}=\mathbf{B} \mathbf{A}$ implies $U_{\mathbf{A} \mathbf{B}} \subseteq U_{\mathbf{A}} \cap U_{\mathbf{B}}$. The preceding inclusions and the dimension formula for sums and intersections now give
    \[
        \begin{aligned}
            \mathrm{r}(\mathbf{A}+\mathbf{B})+\mathrm{r}(\mathbf{A} \mathbf{B}) & =\operatorname{dim} U_{\mathbf{A}+\mathbf{B}}+\operatorname{dim} U_{\mathbf{A} \mathbf{B}} \leq \operatorname{dim}\left(U_{\mathbf{A}}+U_{\mathbf{B}}\right)+\operatorname{dim}\left(U_{\mathbf{A}} \cap U_{\mathbf{B}}\right) \\
                                                                                & =\operatorname{dim} U_{\mathbf{A}}+\operatorname{dim} U_{\mathbf{B}}=\mathrm{r}(\mathbf{A})+\mathrm{r}(\mathbf{B}) .
        \end{aligned}
    \]
\end{question}

\begin{theorem}
    A matrix $\mathbf{A}$ has rank $r$ if and only if it has a nonzero minor $|\mathbf{D}|$ of order $r$ and every bordered minor of order $r+1$ containing $|\mathbf{D}|$ is zero.

    \textbf{Proof.}
    It suffices to prove sufficiency. Without loss of generality, assume that $|\mathbf{D}|$ is the minor formed by the first $r$ rows and first $r$ columns of $\mathbf{A}$. Write
    \begin{equation*}
        \mathbf{A}
        =
        \left(
        \begin{matrix}
                {\alpha}_1 \\
                {\alpha}_2 \\
                \vdots     \\
                {\alpha}_m
            \end{matrix}
        \right)
        =
        \left({\beta}_1, {\beta}_2, \cdots, {\beta}_n\right)
    \end{equation*}
    for the row and column decompositions of $\mathbf{A}$. Let $\tau_{\leq r} {\alpha}_i$ denote the truncation of the row vector ${\alpha}_i$ to its first $r$ entries, and let $\tau_{\leq r} {\beta}_j$ denote the truncation of the column vector ${\beta}_j$ to its first $r$ entries. Since $|\mathbf{D}| \neq 0$, the vectors $\tau_{\leq r} {\alpha}_1, \cdots, \tau_{\leq r} {\alpha}_r$ are linearly independent, and therefore ${\alpha}_1, \cdots, {\alpha}_r$ are linearly independent. It remains to prove that they form a maximal linearly independent set among the rows of $\mathbf{A}$.

    Suppose, to the contrary, that they are not maximal. We may then add a row, say ${\alpha}_{r+1}$, so that ${\alpha}_1, \cdots, {\alpha}_r, {\alpha}_{r+1}$ are linearly independent. Let $\mathbf{A}_1$ be the matrix formed by the first $r+1$ rows of $\mathbf{A}$. Then $\mathbf{A}_1=\left(\tau_{\leq r+1} {\beta}_1, \tau_{\leq r+1} {\beta}_2, \cdots, \tau_{\leq r+1} {\beta}_n\right)$ and $\mathrm{r}\left(\mathbf{A}_1\right)=r+1$. Since $|\mathbf{D}| \neq 0$, the vectors $\tau_{\leq r} {\beta}_1, \cdots, \tau_{\leq r} {\beta}_r$ are linearly independent, as are $\tau_{\leq r+1} {\beta}_1, \cdots, \tau_{\leq r+1} {\beta}_r$. Because $\mathrm{r}\left(\mathbf{A}_1\right)=r+1$, some column of $\mathbf{A}_1$, which we may take to be $\tau_{\leq r+1} {\beta}_{r+1}$, makes $\tau_{\leq r+1} {\beta}_1, \cdots, \tau_{\leq r+1} {\beta}_r, \tau_{\leq r+1} {\beta}_{r+1}$ linearly independent.
    Let $\mathbf{A}_2=\left(\tau_{\leq r+1} {\beta}_1, \cdots, \tau_{\leq r+1} {\beta}_r, \tau_{\leq r+1} {\beta}_{r+1}\right)$. Thus $\mathbf{A}_2$ is the square submatrix formed by the first $r+1$ rows and columns of $\mathbf{A}$, and $\mathrm{r}\left(\mathbf{A}_2\right)=r+1$. Consequently, $\left|\mathbf{A}_2\right| \neq 0$ is a bordered minor of order $r+1$ containing $|\mathbf{D}|$, contradicting the hypothesis.
\end{theorem}

\begin{theorem}
    Let the rows of an $m \times n$ matrix $\mathbf{A}$ be ${\alpha}_1, {\alpha}_2, \cdots, {\alpha}_m$, and suppose ${\alpha}_{i_1}, {\alpha}_{i_2}, \cdots, {\alpha}_{i_r}$ form a maximal linearly independent set. Let its columns be ${\beta}_1, {\beta}_2, \cdots, {\beta}_n$, and suppose ${\beta}_{j_1}, {\beta}_{j_2}, \cdots, {\beta}_{j_r}$ form a maximal linearly independent set. Prove that the submatrix $\mathbf{D}$ at the intersections of these rows and columns satisfies $|\mathbf{D}| \neq 0$.

    \textbf{Proof.}
    Since ${\alpha}_{i_1}, {\alpha}_{i_2}, \cdots, {\alpha}_{i_r}$ are maximal linearly independent, every row ${\alpha}_s$ of $\mathbf{A}$ is a linear combination of them. Let $\widetilde{{\alpha}}_{i_1}, \widetilde{{\alpha}}_{i_2}, \cdots, \widetilde{{\alpha}}_{i_r}, \widetilde{{\alpha}}_s$ be the restrictions of the corresponding rows to columns $j_1, j_2, \cdots, j_r$. Then every $\widetilde{{\alpha}}_s$ is a linear combination of $\widetilde{{\alpha}}_{i_1}, \widetilde{{\alpha}}_{i_2}, \cdots, \widetilde{{\alpha}}_{i_r}$. Now consider $\mathbf{B}=\left({\beta}_{j_1}, {\beta}_{j_2}, \cdots, {\beta}_{j_r}\right)$, an $m \times r$ matrix of rank $r$.
    Every row $\widetilde{{\alpha}}_s$ of $\mathbf{B}$ lies in the span of $\widetilde{{\alpha}}_{i_1}, \widetilde{{\alpha}}_{i_2}, \cdots, \widetilde{{\alpha}}_{i_r}$, and the row rank of $\mathbf{B}$ is $r$. Hence these $r$ rows form a maximal linearly independent set. Thus $\mathbf{D}$ has full rank and $|\mathbf{D}| \neq 0$.
\end{theorem}

    Let $\mathbf{A}$ be an $m \times n$ matrix. Prove the following:

    If $\mathrm{r}(\mathbf{A})=n$, so that $\mathbf{A}$ has full column rank, then there exists an $n \times m$ matrix $\mathbf{B}$ of rank $n$ such that ${\mathbf{B} \mathbf{A}}=\mathbf{I}_n$. Such a matrix $\mathbf{B}$ is called a left inverse of $\mathbf{A}$.

    If $\mathrm{r}(\mathbf{A})=m$, so that $\mathbf{A}$ has full row rank, then there exists an $n \times m$ matrix $\mathbf{C}$ of rank $m$ such that ${\mathbf{A} \mathbf{C}}=\mathbf{I}_m$. Such a matrix $\mathbf{C}$ is called a right inverse of $\mathbf{A}$.

    \textbf{Proof.} (1) Let $\mathbf{P}$ and $\mathbf{Q}$ be invertible matrices of orders $m$ and $n$, respectively, such that
    \begin{equation*}
        \mathbf{P} \mathbf{A} \mathbf{Q}=\binom{\mathbf{I}_n}{\mathbf{O}}
    \end{equation*}

    Therefore, $\left(\mathbf{I}_n, \mathbf{O}\right) \mathbf{P} \mathbf{A} \mathbf{Q}=\mathbf{I}_n$, so $\left(\mathbf{I}_n, \mathbf{O}\right) \mathbf{P} \mathbf{A}=\mathbf{Q}^{-1}$ and hence $\mathbf{Q}\left(\mathbf{I}_n, \mathbf{O}\right) \mathbf{P} \mathbf{A}=\mathbf{I}_n$. It suffices to take $\mathbf{B}=\mathbf{Q}\left(\mathbf{I}_n, \mathbf{O}\right) \mathbf{P}$.
    (2) This follows similarly, or by applying part (1) to $\mathbf{A}^{\prime}$.
    \textbf{Corollary.} A full-column-rank matrix satisfies left cancellation: if $\mathbf{A}$ has full column rank and ${\mathbf{A} \mathbf{D}}={\mathbf{A} \mathbf{E}}$, then $\mathbf{D}=\mathbf{E}$. Similarly, a full-row-rank matrix satisfies right cancellation: if $\mathbf{A}$ has full row rank and ${\mathbf{D} \mathbf{A}}={\mathbf{E} \mathbf{A}}$, then $\mathbf{D}=\mathbf{E}$.


\end{theorem}

\begin{theorem}
    Let $U$ be a $\varphi$-invariant subspace of $V$. Then $U$ is cyclic if and only if the matrix of $\left.\varphi\right|_U$ with respect to some basis of $U$ is the companion matrix of a monic polynomial.
    \begin{proof}
        First prove sufficiency. Suppose the matrix of $\left.\varphi\right|_U$ with respect to a basis $\left\{e_1, e_2, \cdots, e_r\right\}$ of $U$ is the companion matrix $\mathbf{C}(d(\lambda))$, where $d(\lambda)=\lambda^r+a_1 \lambda^{r-1}+\cdots+a_{r-1} \lambda+a_r$. Then
        ${\varphi}\left({e}_i\right)={e}_{i+1}(1 \leq i \leq r-1)$ and ${\varphi}\left({e}_r\right)=-\sum_{i=1}^r a_{r-i+1} {e}_i$. Hence ${e}_i={\varphi}^{i-1}\left({e}_1\right)(2 \leq i \leq r)$, and $U=L\left({e}_1, {e}_2, \cdots, {e}_r\right)=C\left({\varphi}, {e}_1\right)$ is cyclic.

        Now prove necessity. Suppose $U=C({\varphi}, {\alpha})$ is an $r$-dimensional cyclic subspace. Then
        $\{{\alpha}, {\varphi}({\alpha}) \cdots, {\varphi}^{r-1}({\alpha})\}$ is a basis of $U$. Write
        \[
            {\varphi}^r({\alpha})=-a_r {\alpha}-a_{r-1} {\varphi}({\alpha})-\cdots-a_1 {\varphi}^{r-1}({\alpha})
        \]
        Let $d(\lambda)=\lambda^r+a_1 \lambda^{r-1}+\cdots+a_{r-1} \lambda+a_r$. It is immediate that the matrix of $\left.{\varphi}\right|_U$ with respect to the basis $\left\{{\alpha}, {\varphi}({\alpha}), \cdots, {\varphi}^{r-1}({\alpha})\right\}$ is the companion matrix $\mathbf{C}(d(\lambda))$.
    \end{proof}
\end{theorem}

    \[

\end{theorem}

            \psi\left(e_k\right) & =\psi\left({\varphi}^{k-1}\left(e_1\right)\right)={\varphi}^{k-1}\left(\psi\left(e_1\right)\right)={\varphi}^{k-1}\left(g(\varphi)\left(e_1\right)\right) \\

\begin{corollary}

    where $\mathbf{A}_1$ is an $(n-1) \times (n-1)$ matrix over $\mathbb{F}$. Let $\mathbf{P}=\left({e}_1, {e}_2, \cdots, {e}_n\right)$. Then $\mathbf{P}$ is invertible and the preceding equality gives ${\mathbf{A} \mathbf{P}}=\mathbf{P}\left(\begin{array}{cc}\lambda_1 & * \\ \mathbf{O} & \mathbf{A}_1\end{array}\right)$, so $\mathbf{P}^{-1} {\mathbf{A} \mathbf{P}}=\left(\begin{array}{cc}\lambda_1 & * \\ \mathbf{O} & \mathbf{A}_1\end{array}\right)$. It follows that $\left|\lambda \mathbf{I}_n-\mathbf{A}\right|=\left(\lambda-\lambda_1\right)\left|\lambda \mathbf{I}_{n-1}-\mathbf{A}_1\right|$, and hence all eigenvalues of $\mathbf{A}_1$ lie in $\mathbb{F}$. By the induction hypothesis, there exists an invertible $(n-1) \times (n-1)$ matrix $\mathbf{Q}$ over $\mathbb{F}$ such that $\mathbf{Q}^{-1} \mathbf{A}_1 \mathbf{Q}$ is upper triangular. Let
    \[
        \mathbf{R}=\mathbf{P}\left(\begin{array}{ll}
                1          & \mathbf{O} \\
                \mathbf{O} & \mathbf{Q}
            \end{array}\right)
    \]
    Then $\mathbf{R}$ is an invertible $n \times n$ matrix over $\mathbb{F}$, and
    \[
        \mathbf{R}^{-1} \mathbf{A} \mathbf{R}=\left(\begin{array}{cc}
            1          & \mathbf{O} \\
            \mathbf{O} & \mathbf{Q}
        \end{array}\right)^{-1}\left(\begin{array}{cc}
                \lambda_1  & *            \\
                \mathbf{O} & \mathbf{A}_1
            \end{array}\right)\left(\begin{array}{cc}
                1          & \mathbf{O} \\
                \mathbf{O} & \mathbf{Q}
            \end{array}\right)=\left(\begin{array}{cc}
                \lambda_1  & *                                       \\
                \mathbf{O} & \mathbf{Q}^{-1} \mathbf{A}_1 \mathbf{Q}
            \end{array}\right)
    \]
    is upper triangular.
\end{theorem}

\begin{theorem}
    Let $\varphi$ and $\psi$ be linear transformations on an $n$-dimensional vector space $V$ over a field $\mathbb{F}$. Suppose $\varphi \psi=\psi \varphi$ and both $\varphi$ and $\psi$ are diagonalizable. Prove that they are simultaneously diagonalizable.

    \textbf{Proof.}
    We use induction on $\dim V$. The case $n=1$ is immediate. Assume the result for vector spaces of dimension less than $n$. Let the distinct eigenvalues of $\varphi$ be $\lambda_1, \cdots, \lambda_s \in \mathbb{F}$, with corresponding eigenspaces $V_1, \cdots, V_s$. Since $\varphi$ is diagonalizable,
    \[
        V=V_1 \oplus \cdots \oplus V_s .
    \]
    If $s=1$, then ${\varphi}=\lambda_1 \mathbf{I}_V$ is a scalar transformation. Choose a basis of $V$ in which the matrix of ${\psi}$ is diagonal; in the same basis, the matrix of ${\varphi}$ is $\lambda_1 \mathbf{I}_n$. If $s>1$, then $\operatorname{dim} V_i<n$, and every $V_i$ is invariant under $\psi$. The restrictions $\left.\varphi\right|_{V_i}$ and $\left.\psi\right|_{V_i}$ commute and are both diagonalizable. By the induction hypothesis, they are simultaneously diagonalizable on $V_i$. Combining these bases of the $V_i$ gives a basis of $V$ in which both $\varphi$ and $\psi$ are diagonal.
\end{theorem}

\begin{proposition}
    Suppose $\varphi $ is a linear transformation on a finite-dimensional vector space $V$ over a field $\mathbb{F}$, with characteristic polynomial $f(\lambda)$ and minimal polynomial $m(\lambda)$.
    \begin{equation*}
        f(\lambda) = P_1(\lambda)^{r_1} P_2(\lambda)^{r_2} \cdots P_k(\lambda)^{r_k}, m(\lambda) = P_1(\lambda)^{s_1} P_2(\lambda)^{s_2} \cdots P_k(\lambda)^{s_k},
    \end{equation*}
    where $P_i(\lambda)$ are distinct monic irreducible polynomials over $\mathbb{F}$, and $r_i \geq s_i \geq 1$ for all $i$.
    \begin{enumerate}
        \item $V=\bigoplus \ker (P_i(\varphi)^{s_i})$
        \item $\left.\varphi\right|_{V_i}$ has minimal polynomial $P_i(\lambda)^{s_i}$ and characteristic polynomial $P_i(\lambda)^{r_i}$, where $V_i = \ker (P_i(\varphi)^{s_i})$.
    \end{enumerate}
\end{proposition}
